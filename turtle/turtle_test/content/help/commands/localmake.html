; Draws a square of the specified size around current turtle position.
; Does not change turtle position
to square :size
  localmake "originalPos pos
  jumpForward :size * 0.5
  right 90
  jumpBackward :size * 0.5
  ; Now, turtle is at desired heading and position.
  repeat 4 [
    forward :size
    right 90
  ]
  jumpTo :originalPos
  ; Restored original turtle position.
  left 90   ; Restored original heading end

square 100

; Draw squares of various sizes with the same center.
repeat 50 [
  square 100 + repcount * 5
]
