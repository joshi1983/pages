to forwardDashed :len
  if :len < 2 [
    forward :len
    stop
  ]
  forward 2
  localmake "len :len - 2
  if :len < 2 [
    jumpForward :len
    stop
  ]
  jumpForward 2
  forwardDashed :len - 2
end

; A similar procedure except using ifelse instead of stop.
to forwardDashedNoStop :len
  ifelse :len < 2 [
    forward :len
  ] [
    forward 2
    localmake "len :len - 2
    ifelse :len < 2 [
      jumpForward :len
    ] [
      jumpForward 2
      forwardDashedNoStop :len - 2
    ]
  ]
end

forwardDashed 100
right 45 forwardDashedNoStop 75
