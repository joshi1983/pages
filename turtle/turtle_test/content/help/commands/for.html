for ["x 0 5 1] [
  print :x
] ; prints 0 1 2 3 4 5 on separate lines

for ["x 0 5] [   print :x
] ; also prints 0 1 2 3 4 5 on separate lines

for ["x 0 5 0.5] [   print :x
] ; prints 0 0.5 1 1.5 2 2.5 3 3.5 4 4.5 5
for ["x 0 5 2] [   print :x
] ; prints 0 2 4 on separate lines

for ["x 0 -3 -1] [   print :x
] ; prints 0 -1 -2 -3 on separate lines

for ["x 0 -3] [   print :x
] ; also prints 0 -1 -2 -3 on separate lines
