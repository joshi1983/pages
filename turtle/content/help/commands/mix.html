print mix 10 20 0
; prints 20

print mix 10 20 1
; prints 10

print mix 10 20 0.5
; prints 15

print mix 10 20 0.1
; prints 19

print mix 10 20 0.9
; prints 11

print mix [0 10 20] [10 20 30] 0.5
; prints [5 15 25]

print mix "red transparent 0.5
; prints [127.5 255 0 0]
; The 4 element list represents an alphacolor.
; In other words, this prints the
; color red associated with 50% opacity(50% = 127.5 * 100%/255).

print mix "#80ffff00 transparent 0.5
; prints [64 255 255 0]
; 64 = 128(which is 80 in hexadecimal) * 0.5
; 255 = ff in hexadecimal.
