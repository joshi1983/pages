print modulo 4 2
; prints 0

print modulo 2 3
; prints 2

print modulo -2 3
; prints 1

print modulo 2 -3
; prints -1

print modulo -4 2
; prints 0


print remainder -2 3
; prints -2
; modulo would have printed 1 for the same inputs.

print remainder 2 -3
; prints 2
; modulo would have printed -1 for the same inputs.
