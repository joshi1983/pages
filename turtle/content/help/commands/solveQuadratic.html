; The following code graphs the parabola for:
; y = :a * :x * :x + :b * :x + :c.
; It marks the x-axis intersections with red circles.
; solveQuadratic :a :b :c is used
; to calculate the x values.

; adjust the a, b, and c values to see
; the red points adjust correctly.
make "a 2
make "b 5
make "c -50

to quadratic :x
output :a * :x * :x + :b * :x + :c
end

; transforms the x-value to display it more clearly
to mathXToDrawingX :x
output :x * 20
end

; Graph the parabola.
setFillColor "black
repeat 100 [
make "x (repcount - 50) * 0.2
penup
setXY (mathXToDrawingX :x) quadratic :x
pendown
circle 3
]

; Show the solved intersection between ; the parabola and the x-axis.
make "xWhenYIsZero solveQuadratic :a :b :c
setFillColor "red
repeat count :xWhenYIsZero [
make "x item repcount :xWhenYIsZero
penup
setXY (mathXToDrawingX :x) 0
pendown
circle 8
penup
; move away from dot enough to
; make text more readable.
setY yCor + 15
pendown
label str :x
]
