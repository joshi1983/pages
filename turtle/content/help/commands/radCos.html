print radCos 0
; prints 1

print radCos pi
; prints -1

print radCos pi * 0.5
; prints 0

print radCos pi * 0.25
; prints 0.7071068

print cos 45
; also prints 0.7071068
; Note that 45 degrees = pi * 0.25 radians

print radcos radArcCos 1
; prints 1
; radcos and radarccos are inverse functions of each other.

print radcos radArcCos 0.2
; prints 0.2

print radcos radArcCos 0.1
; prints 0.1

print radArcCos radcos 0.1
; also prints 0.1
