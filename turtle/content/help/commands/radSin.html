print radSin 0
; prints 0

print radSin pi
; prints 0

print radSin pi * 0.5
; prints 1

print radSin pi * 0.25
; prints 0.7071068

print sin 45
; also prints 0.7071068
; Note that 45 degrees = pi * 0.25 radians

print radsin radArcSin 1
; prints 1
; radsin and radarcsin are inverse functions of each other.

print radsin radArcSin 0.2
; prints 0.2

print radsin radArcSin 0.1
; prints 0.1

print radArcSin radsin 0.1
; also prints 0.1
