make "p createPList
setproperty "p "x 4
setproperty "p "y 10
print :p
; prints [x:=4 y:=10]

print plist "p
; prints [x 4 y 10]

; The keys don't need to be strings.
; Let's associate 12(not a string) with 25.
setproperty "p 12 25
print :p
; prints [x:=4 y:=10 12:=25]

print plist "p
; prints [x 4 y 10 12 25]
