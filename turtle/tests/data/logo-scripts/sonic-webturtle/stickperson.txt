; Example 6 - Stick Person

; Head:

HOME
MOVE 100
COLOR ORANGE
GO HEAD

; Body:

HOME
MOVE 80
RIGHT 90
COLOR GREEN
GO BODY

; Arms:

HOME
MOVE 80
LEFT 45
COLOR GREEN
GO LEFTARM

HOME
MOVE 80
RIGHT 90
MOVE 30
LEFT 45
COLOR GREEN
GO RIGHTARM

; Legs:

HOME
LEFT 75
COLOR BLUE
GO LEFTLEG

HOME
RIGHT 90
MOVE 30
LEFT 30
COLOR BLUE
GO RIGHTLEG

END

# HEAD
  REPEAT 10
    DRAW 10
    RIGHT 36
  NEXT
RETURN

# BODY
  REPEAT 2
    DRAW 30
    RIGHT 90
    DRAW 80
    RIGHT 90
  NEXT
RETURN

# LEFTARM
  REPEAT 2
    DRAW 15
    LEFT 90
    DRAW 40
    LEFT 90
  NEXT
RETURN

# RIGHTARM
  REPEAT 2
    DRAW 15
    RIGHT 90
    DRAW 40
    RIGHT 90
  NEXT
RETURN

# LEFTLEG
  REPEAT 2
    DRAW 15
    LEFT 90
    DRAW 80
    LEFT 90
  NEXT
RETURN

# RIGHTLEG
  REPEAT 2
    DRAW 15
    RIGHT 90
    DRAW 80
    RIGHT 90
  NEXT
RETURN
