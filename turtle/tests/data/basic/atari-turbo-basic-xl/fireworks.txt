10 GRAPHICS 7+16:SETCOLOR 0,1,14:SETCOLOR 2,2,10:SETCOLOR 1,5,10:DEG
15 FOR t=1 TO 20
20 COLOR RAND(3)+1:W=160:H=96:r=10
30 XC=RAND(W-2*r)-r
40 YC=RAND(H-2*r)-r
50 IF xc>W THEN xc=W-2*r
51 IF x<0 THEN xc=2*r
52 IF yc<0 THEN yc=2*r
53 IF yc>H THEN yc=H-2*r
54 FOR i=0 TO 20
60 PLOT xc,yc
65 x=INT(RAND(r)*COS(i*RAND(360)))+xc
70 y=INT(RAND(r)*SIN(i*RAND(360)))+yc
80 IF x<0 THEN x=0
90 IF y<0 THEN y=0
100 IF x>W THEN x=W
110 IF y>H THEN y=H
120 DRAWTO x,y
130 NEXT i
140 NEXT t
