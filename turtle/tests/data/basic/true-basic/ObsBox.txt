!
!  OBSBOX. Show PlotOBs with polynomial fitting and
!  SqrTrans in two windows.  Note "?" for missing data.
!  This is like OBS but we use box plots instead of error bars.
!  From "Beginning Statistics" by Mosteller et al. p. 333
!
LIBRARY "Stat2Lib.trc"

DIM h$(5),y$(5,4),y(5,4)

CALL SetObsBox (1)                ! Use box plots

MAT READ h$, y$
DATA 20, 30, 40, 50, 60
DATA 12.3,   16.8,   14.5,   ?
DATA 27.0,   34.8,   31.4,  38.4
DATA 70.6,   75.5,   68.9,   ?
DATA 121.0, 130.0,  123.2, 116.6
DATA 179.2, 171.6,  151.3, 166.4
CALL TextToNum2(y$,y)

OPEN #1: screen 0, .49, 0, 1
CALL SetDataStyle (8)
CALL SetPolyFit (2)
CALL SetText ("Stopping", "MPH", "Feet")
CALL PlotObs (y, h$, 0, 0, "")

OPEN #2: screen .5, 1, 0, 1
CALL SetPolyFit (0)
CALL SetLS (1)
CALL SqrTran2 (y)
CALL SetText ("Sqr Root", "MPH", "Sqr Feet")
CALL PlotObs (y, h$, 0, 0, "")

END
