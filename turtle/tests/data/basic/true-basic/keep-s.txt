EXTERNAL

SUB Keep (line$(),arg$)

    LIBRARY "{WB_c}WBkeep.trc"

    WHEN EXCEPTION IN
         CALL WB_Keep (line$(), arg$)
         PRINT "Done with Keep."
    USE
         IF extype = 1 then
            PRINT extext$
         ELSE
            PRINT exline$
            PRINT extype, extext$
            PRINT "Error in Keep."
         END IF
    END WHEN

END SUB
