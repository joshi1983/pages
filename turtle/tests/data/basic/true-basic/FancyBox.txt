!
!  FANCYBOX.  Box plots of the 10 largest cities for 16 countries.
!  Use log-x scale, colored boxes, and cross-bars on whiskers.
!  Taken from "Understanding Robust & Exploratory Data Analysis",
!  pages 66-70.
!
LIBRARY "Stat2Lib.trc"

DIM pop(16,10), names$(16)

MAT READ pop,names$               ! Read population/country names
MAT pop = Trn(pop)                ! Get datasets as columns

CALL SetErrorBeam (3)             ! Put crossbars on whiskers
CALL SetBoxColor ("yellow")       ! Ask for yellow boxes
CALL SetGraphType ("logx")        ! ...and log-x scale

CALL SetText ("10 Largest Cities", "Log Population in 100,000s", "")
CALL SetTickSizes (3, 0)
CALL ManyBoxPlot (pop, names$, "")
