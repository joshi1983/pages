! Illustrate MATLIB with HUGE
!
! **NOTE:** type 'do huge' in the command window before running


DECLARE HUGE a, b, r
Decimals (2)

DIM A(4,3), B(4,3), R(1,1)
MAT READ A, B

MAT R = A - B                     ! Subtract
MAT PRINT R                       ! Print to 2 decimals
MAT R = R + B                     ! Add back in
MAT PRINT R

DATA 1,2,3
DATA 2,3,4
DATA 3,4,5
DATA 4,5,6

DATA 2.13, 1.29, 3.18
DATA -1.23, -2.34, -3.45
DATA 2.13, 1.29, 3.18
DATA -1.23, -2.34, -3.45

END
