!
!  KS.  Show how to use Kolmogorov-Smirnov test.
!  Note that sample sizes aren't big enough for the
!  probability to be reliable.
!
LIBRARY "Stat3Lib.trc"

DIM a(10), b(10)

MAT READ a, b
DATA 2.1, 3.0, 1.2, 2.9, 0.6, 2.8, 1.6, 1.7, 3.2, 1.7
DATA 3.2, 3.8, 2.1, 7.2, 2.3, 3.5, 3.0, 3.1, 4.6, 3.2

CALL KS2 (a, b, d, p)
PRINT "Kolmogorov-Smirnov statistic: "; d

END
