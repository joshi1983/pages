!  MAT PLOT AREA example
!
DIM points (201,2)
SET WINDOW -1, 1, -1, 1

FOR t = 0 to 2 step .01                ! Compute points
    LET c = c+1                        ! Count points
    LET points(C,1) = sin(3*t*pi)      ! x-coordinate
    LET points(c,2) = cos(5*t*pi)      ! y-coordinate
NEXT t

MAT PLOT AREA: points                  ! Draw and fill in

END
