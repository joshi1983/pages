!
!  IRRDATA.  Make and show an "irregular" frequency dataset.
!  Create from simulated normal distribution.
!
LIBRARY "Stat2Lib.trc"

DIM int$(5), d(100), freq(0,0)

MAT READ int$
DATA 0:.3, .3:.45, .45:.55, .55:.7, .7:1

CALL SimNormal (100, .5, .25, d)  ! Create a normal dataset

CALL SetAxes (0)
CALL SetTitle ("An Irregular Frequency Dataset")

CALL DataToIrrFreq (int$(), d(), freq(,))   ! Convert to freq table

CALL PlotHistFromFreq (freq(,), "")    ! Plot histogram

END
