! Complex sines and cosines
!
! **NOTE:** type 'do complex' in the command window before running


DECLARE COMPLEX a, c, s, z
DECIMALS 2
SET ZONEWIDTH 20

PRINT "Number", "Sine", "Cosine", "Sin^2 + Cos^2"
FOR example = 1 to 5
    LET a = 5*rnd - 7*rnd*i
    LET c = cos(a)
    LET s = sin(a)
    LET z = c^2 + s^2
    PRINT a, c, s, z
NEXT example

END
