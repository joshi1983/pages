!
!  FISHER
!
!  Show how to compute Fisher exact probabilities for
!  a 2x2 table.
!
LIBRARY "Stat1Lib.trc"

DIM t(2,2)

MAT READ t                        ! Read 2x2 table
DATA 6, 0, 2, 6

PRINT "For 2x2 Table:"            ! Print on screen
MAT PRINT t;

CALL FisherExact (t, p)           ! Compute probability
PRINT "Fisher exact probability: "; Using$("#.#####",p)

END
