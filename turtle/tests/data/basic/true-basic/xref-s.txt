EXTERNAL

SUB Xref (l$(), options$)

    LIBRARY "{WB_c}WBxref.trc"

    WHEN EXCEPTION IN
         CALL WB_Xref (l$(), options$)
         PRINT "Done with Xref."
    USE
         IF extype = 1 then
            PRINT extext$
         ELSE
            PRINT extype, extext$
            PRINT exline$
            PRINT "Error in Xref."
         END IF
    END WHEN

END SUB
