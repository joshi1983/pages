EXTERNAL

SUB Diff (l$(), options$)

    LIBRARY "{WB_c}WBdiff.trc"

    WHEN EXCEPTION IN
         CALL WB_Diff (l$(), options$)
         PRINT "Done with Diff."
    USE
         IF extype = 1 then
            PRINT extext$
         ELSE
            PRINT extype, extext$
            PRINT exline$
            PRINT "Error in Diff."
         END IF
    END WHEN

END SUB
