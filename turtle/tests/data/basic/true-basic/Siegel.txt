!
!  SIEGEL.  Show how to use Siegel-Tukey rank dispersion test,
!  a nonparametric replacement for the F-test for equal variances.
!  Data from "Applied Statistics" by L. Sachs, p. 288.
!
LIBRARY "Stat3Lib.trc"

DIM a(10), b(10)

MAT READ a, b
DATA 10.1, 7.3, 12.6, 2.4, 6.1, 8.5, 8.8, 9.4, 10.1, 9.8
DATA 15.3, 3.6, 16.5, 2.9, 3.3, 4.2, 4.9, 7.3, 11.7, 13.1

CALL SiegelTukey (a, b, i, z, p)

PRINT using "Siegel Tukey I: ###    Prob(I): #.###": i, p

END
