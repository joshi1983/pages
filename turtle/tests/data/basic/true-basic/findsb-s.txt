EXTERNAL

SUB FindSub (l$(), options$)

    LIBRARY "{WB_c}WBfinds.trc"

    WHEN EXCEPTION IN
         CALL WB_FindSub (l$(), options$)
         PRINT "Done with FindSub."
    USE
         IF extype = 1 then
            PRINT extext$
         ELSE
            PRINT extype, extext$
            PRINT exline$
            PRINT "Error in Findsub."
         END IF
    END WHEN

END SUB
