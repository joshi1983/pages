EXTERNAL

SUB Find (l$(), options$)

    LIBRARY "{WB_c}WBfinds.trc"

    WHEN EXCEPTION IN
         CALL WB_Find (l$(), options$)
         PRINT "Done with Find."
    USE
         IF extype = 1 then
            PRINT extext$
         ELSE
            PRINT extype, extext$
            PRINT exline$
            PRINT "Error in Find."
         END IF
    END WHEN

END SUB
