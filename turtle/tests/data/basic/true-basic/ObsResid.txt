!
!  OBSRESID. Show linear and quadratic residuals for data.
!  From "Beginning Statistics" by Mosteller et al. p. 333
!
LIBRARY "Stat2Lib.trc"

DIM h$(5), y$(5,4), y(5,4)

MAT READ h$,y$
DATA 20, 30, 40, 50, 60
DATA 12.3,   16.8,   14.5,   ?
DATA 27.0,   34.8,   31.4,  38.4
DATA 70.6,   75.5,   68.9,   ?
DATA 121.0, 130.0,  123.2, 116.6
DATA 179.2, 171.6,  151.3, 166.4

CALL TextToNum2 (y$, y)
CALL SetErrorBeam (2)

OPEN #1: screen 0, .49, 0, 1
CALL SetText ("Linear Residuals", "MPH", "Feet")
CALL PlotObsResid (y, h$, 10, 0, "")

OPEN #2: screen .5, 1, 0, 1
CALL SetAutoScale (0, 0)
CALL SetText ("Quadratic Residuals", "MPH", "")
CALL PlotPolyObsResid (y, h$, 2, 10, 0, "")

END
