!
!  WALLIS.  Show Wallis-Moore phase frequency test.
!  Data taken from "Applied Statistics" by L. Sachs, p. 379.
!
LIBRARY "Stat3Lib.trc"

DIM d(22)

MAT READ d
DATA 5,6,2,3,5,6,4,3,7,8,9,7,5,3,4,7,3,5,6,7,8,9

CALL WallisMoore (d(), h, z, p)

PRINT "Phase changes: "; h
PRINT "Test statistic: "; z
PRINT "Probability: "; Using$("#.####",p)

END
