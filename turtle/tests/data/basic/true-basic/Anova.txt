!
!  ANOVA.  Show how to print an ANOVA table.
!  Note use of "?" for missing data. From "Statistical Analysis"
!  by Harnett and Murphy, page 525.
!
LIBRARY "Stat4Lib.trc"

DIM d$(10,3),d(10,3)

MAT READ d$
DATA 130,126,131,  126,127,128,  128,129,133,  132,124,130,  126,126,130
DATA  ? ,130,132,   ? ,125, ? ,   ? ,126, ? ,   ? ,128, ? ,   ? ,126, ?

CALL TextToNum2 (d$, d)           ! Convert data to numbers
CALL PrintAnova (#0, d)           ! Print ANOVA table

END
