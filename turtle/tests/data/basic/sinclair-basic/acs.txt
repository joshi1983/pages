10 RUN AT 3: LAYER 0: BORDER 0: PAPER 0: INK 7: CLS
20 LOAD "ACS and Flames.spr" BANK 12
30 SPRITE BANK 12: SPRITE PRINT 1: SPRITE BORDER 1:%a=3:%x=64:%y=112:d=24
35 FOR q=1 TO 6:w=(2+INT(RND*30)/2)
40 SPRITE q,160+(q*16),32,3,9: SPRITE CONTINUE q,160+(q*16),32 TO 192 STEP w RUN ,3 TO 4,1,3
45 NEXT q
47 SPRITE 8,16,16,0,1: SPRITE -9,16,0,1,1: SPRITE -10,32,0,2,1
48 SPRITE CONTINUE 8,16 TO 256 STEP 2 RUN ,16
50 REPEAT : SPRITE 0,%x,%y,%a,1
60 %a=%a+1: IF %a > 4 THEN %a=3
70 IF INKEY$ ="p" THEN IF %x < 160 THEN %x=%x+2
80 IF INKEY$ ="o" THEN IF %x >32 THEN %x=%x-2
90 IF INKEY$ ="q" THEN IF %y >32 THEN %y=%y-2
100 IF INKEY$ ="a" THEN IF %y < 192 THEN %y=%y+2
105 PROC FLAMES()
110 SPRITE MOVE INT : REPEAT UNTIL 0
120 DEFPROC FLAMES()
130 SPRITE 24,64,64,d,1
140 SPRITE 25,64,80,d+16,1
150 SPRITE 26,64,96,d+32,1
155 d=d+1: IF d=32 THEN d=24
157 SPRITE MOVE INT: SPRITE MOVE INT
160 ENDPROC
