5 PAPER 0: CLS : INK 4: DIM a(4): DIM b(4): DIM c(4): DIM d(4): DIM e(4): DIM f(4): DIM p(4)
10 FOR x=1 TO 4: READ a(x): NEXT x: FOR x=1 TO 4: READ b(x): NEXT x
15 FOR x=1 TO 4: READ c(x): NEXT x: FOR x=1 TO 4: READ d(x): NEXT x
20 FOR x=1 TO 4: READ e(x): NEXT x: FOR x=1 TO 4: READ f(x): NEXT x
25 FOR x=1 TO 4: READ p(x): NEXT x: LET xs=15: LET ys=14: LET xo=45: LET yo=10: LET x=0: LET y=0
30 DATA 0,.20,-.15,.85: DATA 0,-.26,.28,.04: DATA 0,.23,.26,-.04: DATA .16,.22,.24,.85
35 DATA 0,0,0,0: DATA 0,1.6,.44,1.6: DATA 328,2621,4925,32767
40 FOR i=1 TO 7000: LET j=RND*32767
45 IF j<p(1) THEN LET k=1: GO TO 65
50 IF j<p(2) THEN LET k=2: GO TO 65
55 IF j<p(3) THEN LET k=3: GO TO 65
60 LET k=4
65 LET nx=a(k)*x+b(k)*y+e(k): LET y=c(k)*x+d(k)*y+f(k): LET x=nx
70 PLOT x*xs+xo,y*ys+yo: PLOT 255-(x*xs+xo),y*ys+yo
75 NEXT i: INK 6: PRINT AT 5,10;"ZX SPECTRUM";AT 6,14;"FERNS"
