2 LET scrw=255: LET scrh=175
4 LET sw=127: LET sh=87
6 BORDER 1: PAPER 1: INK 7 : CLS

LET Y=(B-sh)/sc

LET D=ABS (X)
18 IF ABS (Y) <D THEN
LET D=ABS (Y)
20 GO TO 12
22 IF D<.01 THEN
INK 6:
PLOT A,B:
