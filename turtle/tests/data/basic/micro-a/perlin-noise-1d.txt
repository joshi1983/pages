'perlin noise
mode 1 : wcolor 0,0,0 : swap
var p[512],i,x,y,arg1d,xval
i=1
WHILE i < 256
	p[i] = RAND(256)
	p[i + 256] = p[i]
	i = i + 1
WEND
FCOLOR 230,200,100
print 10,10, "Perlin 1D" : swap
x=1
While x < 640
	arg1d = x / 40
	xval = sin((arg1d*arg1d))
	y = 120 + xval * 70
	FCOLOR 230,200,100
	Print x, y, ".": swap
	x=x+1
Wend
