    10  REM RADIAL
    20  HGR2 : PI = 3.14 : X = 20
    30  A = 2 * PI / X : C = 1 / SIN (PI / X)
    40  S = 140 : T = 95
    50  FOR L = 1 TO X + 0.5
    60    M = 0 : N = 0 : R = (L - 1) * A
    70    FOR Q = 1 TO X
    80      M = M + COS (R) / C
    90      N = N + SIN (R) / C
    100     U = 140 + INT (95 * M + 0.5)
    110     V = 95 + INT (95 * N + 0.5)
    120     HPLOT S,T TO U,V
    130     S = U : T = V : R = R + A
    140 NEXT : NEXT
