10 HGR
20 R=70
30 PI = 3.1415927
40 FOR I = 0 TO 24
50 A1 = 2*PI/100 * I
60 X1 = R * COS(A1)
70 Y1 = R * SIN(A1)
80 FOR J = I+1 TO 24
90 A2 = 2*PI/100 * J
100 X2 = R * COS(A2)
110 Y2 = R * SIN(A2)
120 HPLOT X1,Y1 TO X2,Y2
130 NEXT J
140 NEXT I
] RUN
