    10 REM LINE MOSAIC
    20 GR: COLOR=3 : HOME
    30 X = 19: Y = 20: D = 1
    40 FOR S = 1 TO 37
    50   IF D = 1 THEN HLIN X, X + S AT Y: X = X + S
    60   IF D = 2 THEN VLIN Y, Y + S AT X: Y = Y + S
    70   IF D = 3 THEN HLIN X - S, X AT Y: X = X - S
    80   IF D = 4 THEN VLIN Y - S, Y AT X: Y = Y - S
    90   D = D + 1
    100  IF D > 4 THEN D = 1
    110 NEXT S
    120 END
