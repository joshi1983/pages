1 PR#0:TEXT:HOME
10 REM HYSTERISUSLUS
20 HGR2
30 CX = 140 : CY = 96
40 SX = 120 : SY = 80
50 REM ---- ASSEN ----
60 HCOLOR = 3
70 FOR X = 0 TO 279 : HPLOT X, CY : NEXT
80 FOR Y = 0 TO 191 : HPLOT CX, Y : NEXT
90 REM ANIMATIEPAR
100 PH = 0.30
110 DPH = 0.01
120 SAT = 1.8
130 AMP = 1.0
140 REM ANIMATIE
150 FOR FRAME = 1 TO 2000
160 REM FADING
170 HCOLOR = 1
180 F = 0
190 FOR T = 0 TO 6.28 STEP 0.04
200 H = SIN(T)
210 S1 = SIN(T - PH)
220 B = AMP * (S1 / (1 + SAT * S1 * S1))
230 PX = CX + H * SX
240 PY = CY - B * SY
250 IF F = 0 THEN HPLOT PX,PY : F = 1
260 IF F = 1 THEN HPLOT TO PX,PY
270 NEXT
280 REM NIEUWE LUS - HELDERE KLEUR
290 HCOLOR = 5
300 F = 0
310 FOR T = 0 TO 6.28 STEP 0.02
320 H = SIN(T)
330 S1 = SIN(T - PH)
340 B = AMP * (S1 / (1 + SAT * S1 * S1))
350 PX = CX + H * SX
360 PY = CY - B * SY
370 IF F = 0 THEN HPLOT PX,PY : F = 1
380 IF F = 1 THEN HPLOT TO PX,PY
390 NEXT
400 REM FASEVERSCHUIVING
410 PH = PH + DPH
420 IF PH > 0.60 THEN DPH = -0.01
430 IF PH < 0.20 THEN DPH = 0.01
440 NEXT FRAME
450 END
