100 PAGE
110 SET DEGREES
120 PRINT "MULTIPLIER ";
130 INPUT M
140 MOVE 65,50
150 FOR A=3 TO 7200 STEP 3
160 R0=2+A/160
170 R=SIN(M*A)
180 X=65+R0*R*COS(A)
190 Y=50+R0*R*SIN(A)
200 DRAW X,Y
210 NEXT A
220 HOME
230 PRINT
240 PRINT
