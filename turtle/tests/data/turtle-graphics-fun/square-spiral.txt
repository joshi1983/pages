function square(side) {
  repeat(4, function () {
    forward(side);
    right(90);
  });
}

function demo() {
  hideTurtle();
  color("blue");
  var s = 100;
  while (s > 0) {
    square(s);
    right(36);
    s = s - 10;
  }
}
