def waves = Picture.fromVertexShape { 
  x =>
  x.beginShape()
  x.vertex(-100, 0)
  x.bezierVertex(
    -80, 90, 
    -30, -90, 
    10, 0
    )
 x.endShape()
 }
  
def collection(count:Long):Picture = 
  if (count==1) {
    waves
  } else {
def iHave = waves
  .thatsFilledWith(gradients)
  .thatsStrokeColored(Color(0, 2, 255))
  .thatsScaled(2.0,0.1)
  .thatsTranslated(-365, -60)
  
def iGot = collection(count - 1)
  picStack(
    iHave,
    iGot.thatsTranslated(0, -10)
  )
}
