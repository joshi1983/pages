cleari()
setSpeed(superFast)
setPenThickness(3)
setBackground(randomColor)
setPenColor(randomColor)
zoom(0.6)
repeatFor(rangeTo(1, 150)) { e =>
    repeat(8) {
        forward(e)
        right(50 * 2 + 60)
    }
    right(180)
}
