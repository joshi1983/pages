cleari()
setSpeed(fast)
setAnimationDelay(10)
setBackground(black)
setPenThickness(1.2)
zoom(1.2)
def tree(n: Int) {
    if (n < 10) {
        setFillColor(cm.gold)
        circle(3)
    }
    else {
        forward(n)
        setFillColor(cm.gold)
        circle(3)
        left(20)
        tree(3 * n / 4)
        right(50)
        tree(3 * n / 4)
        left(30)
        hop(-n)
    }
}
hop(-50)
back(110)
tree(110)
setPosition(-195, -160)
setPenColor(yellow)
setPenFont(Font("Quicksand Medium", 20))
write("Draw Fractal Tree Using Recursive Command")
setPenColor(white)
setPosition(-196, -161)
write("Draw Fractal Tree Using Recursive Command")
invisible()
