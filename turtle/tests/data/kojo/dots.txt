clear()
setBackground(black)
setPenThickness(2)

repeat(500) {
  val angle = random(360)
  val dist = random(250)

  val x = dist * math.cos(math.toRadians(angle))
  val y = dist * math.sin(math.toRadians(angle))

  setPenColor(Color(random(255), random(255), random(255)))
  setPosition(x, y)   // FIXED
  dot(3)
}
