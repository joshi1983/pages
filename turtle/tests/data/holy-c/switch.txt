I64 i = 10;
switch (i) {
  case:
    "is 0\n";
    break;
  case:
    "is 1\n";
    break;
  case 4:
    "is four\n";
    break;
  case 5 ... 10:
    "between 5 and ten\n";
    break;
  case:
    "value is 11\n";
    break;
  default:
    "Default case\n";
    break;
}
