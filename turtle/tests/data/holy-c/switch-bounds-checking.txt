I64 i = 0;
switch [i] { /* <- Use square braces to use a nobounds switch*/
  case:
    "is 0\n";
    break;
  case:
    "is 1\n";
    break;
  case 4:
    "is four\n";
    break;
  case 5 ... 10:
    "between 5 and ten\n";
    break;
}
