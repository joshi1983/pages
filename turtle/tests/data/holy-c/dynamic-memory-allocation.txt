U64 *arr = MAlloc(sizeof(U64) * 5);

/* assign 24 to the first element of the heap allocated array. */
arr[0] = 24;


/* Re allocating the array to store 10 unsigned 64bit integers. */
U64 tmp = ReAlloc(arr,sizeof(U64)*10);
if (tmp != NULL) {
  arr = tmp;
} else {
  "Failed to allocate memory!\n";
}
Free(tmp);
